void Merge(int LA[], int LB[], int LC[], int m, int n) {
int i = 0, j = 0, k = 0;
while (i < m && j < n) {
if (LA[i] <= LB[j])
LC[k++] = LA[i++];
else
LC[k++] = LB[j++];
}
while (i < m) LC[k++] = LA[i++];
while (j < n) LC[k++] = LB[j++];
}

Node* Merge(Node* LA, Node* LB) { // LA, LB 带 dummy head
Node* LC = (Node*)malloc(sizeof(Node)); // dummy head
Node* tail = LC;
LA = LA->next; LB = LB->next; // 跳过 dummy head
while (LA && LB) {
if (LA->data <= LB->data) {
tail->next = LA;
LA = LA->next;
} else {
tail->next = LB;
LB = LB->next;
}
tail = tail->next;
}
tail->next = LA ? LA : LB;
return LC;
}
